#include<iostream>
#include<forward_list>
using namespace std;

int main(){
    //intialise forward list with {1,2,3,4}
    forward_list<int> fl = {1,2,3,4};
    //insert 5 at the front
    fl.push_front(5);
    //insert 6 at the front
    fl.push_front(6);
    // insert 7 at position 3
    auto it = fl.begin();
    advance(it,1);
    fl.insert_after(it, 7);
    //print the forward list
    for(auto i:fl){
        cout<<i<<" ";
    }
    cout<<endl;
    //remove the first element
    fl.pop_front();
    // remove the 4th element
    it = fl.begin();
    advance(it,2);
    fl.erase_after(it);
    //print the forward list
    for(auto i:fl){
        cout<<i<<" ";
    }
    cout<<endl;

}
